Por otra parte es necesario buscar en la table NMTIPOSTRANSACCIONES las transacciones marcadas con la conlumna EsIncentivo = 1, todos estos valores los debemos sumar al salario del empleado para fines de calculo de impuesto sobre la renta

Esta es la forma como consulto mi table para fines de calculo de impuesto
DECLARE @SalarioAnual DECIMAL(18,2) = 720000;
DECLARE @Ano INT = 2026;

SELECT TOP 1
       SueldoInicial,
       SueldoFinal,
       Base,
       Valor
FROM NMISR
WHERE YEAR(FechaInicial) = @Ano
  AND @SalarioAnual >= SueldoInicial
  AND (
        SueldoFinal IS NULL
        OR @SalarioAnual <= SueldoFinal
      );

Y este es un aproximado del calculo, DECLARE @SalarioAnual DECIMAL(18,2) = 720000;
DECLARE @Ano INT = 2026;

SELECT
    @SalarioAnual AS SalarioAnual,
    Base,
    Valor,
    Base + ((@SalarioAnual - SueldoInicial) * (Valor / 100.00)) AS ISRAnual,
    (Base + ((@SalarioAnual - SueldoInicial) * (Valor / 100.00))) / 12 AS ISRMensual
FROM ISR
WHERE YEAR(FechaInicial) = @Ano
  AND @SalarioAnual >= SueldoInicial
  AND (
        SueldoFinal IS NULL
        OR @SalarioAnual <= SueldoFinal
      );
